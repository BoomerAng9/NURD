export default function Home() {
  const [form, setForm] = useState({
    first_name: '',
    age: '',
    grade_level: '',
    user_type: '',
    gender: '',
    path_choice: ''
  })

  const handleChange = (e) => {
    const { name, value } = e.target
    setForm((prev) => ({ ...prev, [name]: value }))
  }

  const handleSubmit = async (e) => {
    e.preventDefault()
    const { error } = await supabase.from('users').insert([form])
    if (error) alert('Error: ' + error.message)
    else window.location.href = '/dashboard'
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-purple-600 via-blue-500 to-orange-400 flex items-center justify-center text-white px-4">
      <form onSubmit={handleSubmit} className="bg-white bg-opacity-90 p-6 rounded-lg space-y-4 w-full max-w-md text-black">
        <h1 className="text-3xl font-bold text-center text-indigo-700">Become a NURD</h1>

        <input className="input" name="first_name" placeholder="First Name" required onChange={handleChange} />
        <input className="input" name="age" type="number" placeholder="Age" required onChange={handleChange} />
        <input className="input" name="grade_level" placeholder="Grade Level" required onChange={handleChange} />
        
        <select className="input" name="user_type" required onChange={handleChange}>
          <option value="">Parent or Student?</option>
          <option value="parent">Parent</option>
          <option value="student">Student</option>
        </select>

        <select className="input" name="gender" required onChange={handleChange}>
          <option value="">Select Gender</option>
          <option value="male">Male</option>
          <option value="female">Female</option>
        </select>

        <input className="input" name="path_choice" placeholder="Growth Path (optional)" onChange={handleChange} />

        <button type="submit" className="w-full bg-indigo-600 text-white p-2 rounded">
          Start Your VIBE
        </button>
      </form>
    </div>
  )
}
